 22 /**
 23  * Makes a copy of the arguments from begin to end.
 24  *
 25  * \param begin The begining of the range.
 26  * \param end The end of the range.
 27  * \param args The arguments.
 28  *
 29  * \return The remaining arguments.
 30  */
 31 Array.prototype.slice_args = function (args, begin, end) {
 32   return Array.prototype.slice.call (args, begin, end);
 33 }
 34 
 35 /**
 36  * Applies action to all of the elements of the Array.
 37  *
 38  * \param action The action to apply.
 39  *
 40  * \return The Array.
 41  */
 42 Array.prototype.apply = function (action) {
 43   var args = Array.prototype.slice_args (arguments);
 44   args.unshift (this.length);
 45   args.unshift (0);
 46   return this.apply_range.apply (this, args);
 47 }
 48 
 49 /**
 50  * Applies action to the specified range of the Array elements.
 51  *
 52  * \param begin The beginning of the range.
 53  * \param end The end of the range.
 54  * \param action The action to apply.
 55  *
 56  * \return The Array.
 57  */
 58 Array.prototype.apply_range = function (begin, end, action) {
 59   var args = Array.prototype.slice_args (arguments, 3);
 60   for (var i = begin; i < end; ++i) {
 61     args.unshift (this[i]);
 62     this[i] = action.apply (null, args);
 63     args.shift ();
 64   }
 65   return this;
 66 }

 68 /**
 69  * Applies action to the specified indices of the Array elements.
 70  *
 71  * \param The indices to which to apply the action.
 72  * \param action The action to apply.
 73  *
 74  * \return The Array.
 75  */
 76 Array.prototype.apply_index = function (indices, action) {
 77   var args = Array.prototype.slice_args (arguments, 2);
 78   for (var i = 0, length = indices.length; i < length; ++i) {
 79     args.unshift (this[indices[i]]);
 80     this[indices[i]] = action.apply (null, args);
 81     args.shift ();
 82   }
 83   return this;
 84 }

 86 /**
 87  * Runs action for all of the elements of the Array.
 88  *
 89  * \param action The action to apply.
 90  *
 91  * \return The Array.
 92  */
 93 Array.prototype.for_each = function (action) {
 94   var args = Array.prototype.slice_args (arguments);
 95   args.unshift (this.length);
 96   args.unshift (0);
 97   return this.for_each_range.apply (this, args);
 98 }

100 /**

106  *
107  * \return The Array.
108  */
